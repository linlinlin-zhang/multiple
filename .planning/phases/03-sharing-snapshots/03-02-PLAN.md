```typescript
export default defineConfig({
  plugins: [react()],
  resolve: {
```

```javascript
if (req.method === "GET") {
  if (url.pathname === "/history" || url.pathname === "/history/") {
    return serveStatic("/history/index.html", res);

    return serveStatic("/share/index.html", res);

  return serveStatic(url.pathname, res);
}
```
