# Plan 06: E2E Verification

## Objective
Create and run an end-to-end verification script that covers at least 12 chat→canvas scenarios. The script calls the `/api/chat` endpoint directly (Node.js HTTP client, no browser automation) and asserts that the correct action types are returned for natural-language inputs. This validates the entire chain: prompt → LLM → tool calling → action extraction → normalization → response.

## must_haves
- ≥ 12 test scenarios covering all 7 rich node types, zoom actions, selection, chat-only, and fallback
- Script exits 0 when all tests pass, exits 1 with detailed failure report when any test fails
- Tests run against the local dev server (`npm run dev`)
- Each test asserts both: (a) correct action type returned, (b) non-empty reply present
- Failed tests show the actual vs expected action type and the raw reply for debugging

Create `scripts/test-chat-canvas.js` with the following structure:

```js
#!/usr/bin/env node
// scripts/test-chat-canvas.js
// E2E verification for Phase 21 Chat-to-Canvas reliability

const BASE_URL = process.env.BASE_URL || "http://localhost:3000";
const TIMEOUT_MS = 60000;

async function postChat(message, opts = {}) {
  const res = await fetch(`${BASE_URL}/api/chat`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      message,
      stream: false,
      thinkingMode: "no-thinking",
      language: "zh",
      sessionId: `test-${Date.now()}`,
      ...opts
    })
  });
  if (!res.ok) throw new Error(`HTTP ${res.status}: ${await res.text()}`);
  return res.json();
}

function assertAction(data, expectedType, testName) {
  const actions = data.actions || [];
  const hasAction = actions.some(a => a.type === expectedType);
  if (!hasAction) {
    console.error(`FAIL: ${testName}`);
    console.error(`  Expected action: ${expectedType}`);
    console.error(`  Actual actions: ${actions.map(a => a.type).join(", ") || "(none)"}`);
    console.error(`  Reply preview: ${(data.reply || "").slice(0, 200)}`);
    return false;
  }
  if (!data.reply || data.reply.length < 3) {
    console.error(`FAIL: ${testName} — reply too short or empty`);
    return false;
  }
  console.log(`PASS: ${testName}`);
  return true;
}

async function run() {
  let passed = 0;
  let failed = 0;

  // Test cases defined below
  // ...

  console.log(`\nResults: ${passed} passed, ${failed} failed`);
  process.exit(failed > 0 ? 1 : 0);
}

run().catch((e) => { console.error(e); process.exit(1); });
```

```js
const tests = [
  { message: "做个去日本7天计划", expect: "create_plan", name: "Rich node: plan" },
  { message: "列个今天的任务清单", expect: "create_todo", name: "Rich node: todo" },

  { message: "打开上海地图", expect: "create_map", name: "Rich node: map" },
  { message: "保存这个链接 https://example.com", expect: "create_link", name: "Rich node: link" },
  { message: "写个快速排序代码", expect: "create_code", name: "Rich node: code" },
  { message: "放大画布", expect: "zoom_in", name: "Zoom: zoom_in" },
  { message: "重置视图", expect: "reset_view", name: "Zoom: reset_view" },
  { message: "选中 analysis 卡片", expect: "select_analysis", name: "Selection: select_analysis" },
  { message: "随便聊聊，今天怎么样", expect: null, name: "Chat-only: no action" },
  { message: "帮我规划一下下周工作", expect: "create_plan", name: "Fallback: plan keyword" },
  { message: "写个待办事项", expect: "create_todo", name: "Fallback: todo keyword" },
];

for (const test of tests) {
  try {
    const data = await postChat(test.message);
    if (test.expect === null) {
      // Chat-only: assert NO actions
      const actions = data.actions || [];
      if (actions.length === 0 && data.reply?.length > 3) {
        console.log(`PASS: ${test.name}`);
        passed++;
      } else {
        console.error(`FAIL: ${test.name} — expected no actions, got ${actions.map(a=>a.type).join(", ")}`);
        failed++;
      }
    } else {
      if (assertAction(data, test.expect, test.name)) passed++; else failed++;
    }
  } catch (e) {
    console.error(`FAIL: ${test.name} — ${e.message}`);
    failed++;
  }
}
```

Note: Some tests may be flaky depending on LLM behavior. For the initial run, mark known-flaky tests with a retry loop (max 2 retries) or log warnings instead of hard failures. The goal is to verify the plumbing works, not to train the model.

For "Chat-only" tests, also assert that the reply is conversational (contains at least 10 characters).
