## Existing node structure
Nodes are DOM elements with class `.node`. Each node has:
- `.option-title` or `h3` for the title text (option nodes have `.option-title`, generated nodes have `h3`, analysis node has `h2`)
- `.collapse-dot` button prepended inside the node
- Node data stored in `state.nodes` Map: `{ id, element, x, y, width, height, option, generated, imageHash, explanation }`
- `state.links` array connects parent to child: `{ from, to, kind }`

## Existing delete guards logic
`getChildren(id)` returns direct children from `state.links`.
`getDescendants(id)` returns all descendants recursively.
Source node id is `"source"`.

         if (!canDeleteNode(nodeId)) {
           const reason = nodeId === "source" ? t("node.cannotDeleteSource") : t("node.cannotDeleteWithChildren");
           showSelectionToast(reason); // reuse toast from 09-02
           return;

         const node = state.nodes.get(nodeId);
         if (!node) return;

         // Deselect if this node was selected
         if (state.selectedNodeId === nodeId) {
           deselectNode();
         }

         // Remove from state
         state.nodes.delete(nodeId);
         state.collapsed.delete(nodeId);
         state.selectiveHidden.delete(nodeId);

         // Remove links connected to this node
         state.links = state.links.filter(l => l.from !== nodeId && l.to !== nodeId);

         // Remove DOM element
         node.element.remove();

         // Update counts and UI
         if (node.generated) {
           state.generatedCount = Math.max(0, state.generatedCount - 1);
         }
         updateCounts();
         updateCollapseControls();
         drawLinks();
         autoSave();
       }

    1. In `public/app.js`, create `makeTitleEditable(nodeId, titleElement)`:
       ```javascript
       function makeTitleEditable(nodeId, titleElement) {
         if (!titleElement) return;

         titleElement.addEventListener("dblclick", (event) => {
           event.stopPropagation();
           if (titleElement.querySelector(".node-title-input")) return; // already editing

           const originalText = titleElement.textContent;
           const input = document.createElement("input");
           input.type = "text";
           input.className = "node-title-input";
           input.value = originalText;

           // Replace title text with input
           titleElement.textContent = "";
           titleElement.appendChild(input);
           input.focus();
           input.select();

           function save() {
             const newText = input.value.trim();
             if (newText && newText !== originalText) {
               titleElement.textContent = newText;
               // Update node data
               const node = state.nodes.get(nodeId);
               if (node) {
                 if (node.option) node.option.title = newText;
                 // Also update any other places that reference the title
                 autoSave();
               }
             } else {
               titleElement.textContent = originalText;
             }
           }

           function cancel() {
             titleElement.textContent = originalText;
           }

           input.addEventListener("keydown", (e) => {
             if (e.key === "Enter") {
               e.preventDefault();
               save();
             } else if (e.key === "Escape") {
               e.preventDefault();
               cancel();
             }
           });

           input.addEventListener("blur", () => {
             save();
           });
         });
       }
       ```
