       });
     }

     ```javascript
     async function init() {
       // ... existing init code ...
       const urlParams = new URLSearchParams(window.location.search);
       const resumeSessionId = urlParams.get("session");
       if (resumeSessionId) {
         try {
           await loadSession(resumeSessionId);
         } catch {
           const url = new URL(window.location.href);
           url.searchParams.delete("session");
           window.history.replaceState({}, "", url);
         }
       }
     }
     // Change init(); to init().catch(console.error); at the bottom
     ```
   - This ensures a broken session ID does not leave the user stuck with a bad URL.
