}

Create `src/api/sessions.js` with three handlers. ALL database writes MUST use `prisma.$transaction([...])`.

### Helper: serialize state to DB shape
```javascript
function serializeState(state, sessionId = undefined) {
  // Extract nodes from state.nodes Map
  const nodeEntries = Array.from(state.nodes.values());
  const nodes = nodeEntries.map((n) => ({
    nodeId: n.id,
    type: n.id === "source" ? "source" : n.id === "analysis" ? "analysis" : n.generated ? "generated" : "option",
    x: n.x,
    y: n.y,
    width: n.width || 318,
    height: n.height || 220,
    data: n.option ? { option: n.option, imageHash: n.imageHash || null } : n.id === "source" ? { fileName: state.fileName } : n.id === "analysis" ? { summary: state.latestAnalysis?.summary, detectedSubjects: state.latestAnalysis?.detectedSubjects, moodKeywords: state.latestAnalysis?.moodKeywords } : {},
    collapsed: state.collapsed.has(n.id)
  }));

  const links = (state.links || []).map((l) => ({
    fromNodeId: l.from,
    toNodeId: l.to,
    kind: l.kind || "option"
  }));

  const chatMessages = (state.chatMessages || []).map((m) => ({
    role: m.role,
    content: m.content
  }));

  return { nodes, links, chatMessages };
}
```

### Handler: handleCreateSession(body, res)
1. Accept `body.state` (the full client state object) and optional `body.title`
2. Determine `isDemo` from `body.isDemo` or default `false`
3. Inside `prisma.$transaction(async (tx) => { ... })`:
   a. Create Session: `tx.session.create({ data: { title: body.title || "未命名会话", isDemo, viewState: body.state.view || { x: 0, y: 0, scale: 0.86 } } })`
   b. If `body.state.sourceImage` is a data URL, call `storeDataUrl(body.state.sourceImage, { kind: "upload" })` OUTSIDE the transaction (file writes cannot be rolled back), then pass the hash into the transaction to create the Asset row: `tx.asset.create({ data: { sessionId: session.id, hash: result.hash, kind: "upload", mimeType: result.mimeType, fileSize: result.size, fileName: body.state.fileName } })`
   c. For each generated node in `body.state.nodes` that has an image data URL, similarly store the data URL as `kind: "generated"` outside the transaction, then create Asset rows inside the transaction.
   d. Create all Nodes: `tx.node.createMany({ data: nodes.map(n => ({ ...n, sessionId: session.id })) })`
   e. Create all Links: `tx.link.createMany({ data: links.map(l => ({ ...l, sessionId: session.id })) })`
   f. Create all ChatMessages: `tx.chatMessage.createMany({ data: chatMessages.map(m => ({ ...m, sessionId: session.id })) })`
4. Return `{ ok: true, sessionId: session.id, savedAt: session.updatedAt }`

### Handler: handleGetSession(sessionId, res)
1. Validate `sessionId` is a non-empty string
2. Query: `prisma.session.findUnique({ where: { id: sessionId }, include: { nodes: true, links: true, assets: true, chatMessages: { orderBy: { createdAt: "asc" } } } })`
3. If not found: 404
4. Return the full session object as JSON

### Handler: handleUpdateSession(sessionId, body, res)
1. Validate `sessionId`
2. Inside `prisma.$transaction(async (tx) => { ... })`:
   a. `tx.node.deleteMany({ where: { sessionId } })`
   b. `tx.link.deleteMany({ where: { sessionId } })`
   c. `tx.chatMessage.deleteMany({ where: { sessionId } })`
   d. Re-create nodes, links, chatMessages from `body.state` (same logic as create)
   e. `tx.session.update({ where: { id: sessionId }, data: { title: body.title, updatedAt: new Date(), viewState: body.state.view } })`
3. Handle new assets (sourceImage or generated images that weren't stored before) using the same outside-transaction file storage pattern.
4. Return `{ ok: true, sessionId, savedAt }`

### Error handling
- All handlers wrap in try/catch
- On error: `sendJson(res, 500, { error: error.message })` (or 400 for validation errors)
- Log errors with `console.error`

### Handler: handleListHistory(query, res)
1. Accept optional query params (parsed from URL by server.js):

   - `?includeDemo=false` (default false; if true, include demo sessions)
2. Build Prisma query:
```javascript
const where = {};
if (query.includeDemo !== "true") {
  where.isDemo = false;
}

const sessions = await prisma.session.findMany({
  where,
  orderBy: { updatedAt: "desc" },
  skip: Number(query.offset) || 0,
  take: Math.min(Number(query.limit) || 20, 100),
  select: {
    id: true,
    title: true,
    isDemo: true,
    createdAt: true,
    updatedAt: true,
    _count: {
      select: { nodes: true, assets: true }
    }
  }
});
```
3. Return JSON:
```json
{
  "ok": true,
  "sessions": [
    {
      "id": "...",
      "title": "...",
      "isDemo": false,
      "createdAt": "2026-04-25T...",
      "updatedAt": "2026-04-25T...",
      "nodeCount": 5,
      "assetCount": 2
    }
  ],
  "total": 42
}
```
4. `total` is from `prisma.session.count({ where })`

### Design rationale

- Ordered by `updatedAt DESC` per PERS-02
- Demo sessions excluded by default per DATA-04

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-05 | Tampering | `handleUpdateSession` | mitigate | Only the session owner (in single-user context, any client) can update; no auth yet, but UUID IDs are unguessable |
| T-01-06 | Denial of Service | `handleListHistory` | mitigate | `take` clamped to max 100; `skip` is numeric; prevents unbounded queries |
| T-01-07 | Information Disclosure | `handleGetSession` | mitigate | Returns all session data to any requester; acceptable for single-user local use, but UUID prevents enumeration |
| T-01-08 | Repudiation | `handleCreateSession` | mitigate | `prisma.$transaction` ensures all-or-nothing write; no partial session corruption |
</threat_model>
